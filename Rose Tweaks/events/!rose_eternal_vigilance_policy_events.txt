namespace = rose_eternal_vigilance_policy

country_event = {
	id = rose_eternal_vigilance_policy.1
	hide_window = yes

	is_triggered_only = yes

	trigger = {
		has_ascension_perk = ap_eternal_vigilance
	}

	immediate = {
		if = {
			limit = {
				has_policy_flag = rose_eternal_vigilance_policy_full_spending
			}
			every_owned_starbase = {
				limit = {
					fleet = { is_in_combat = no }
					is_controlled_by = owner
					used_defense_platform_capacity_percent < 0.33
				}
				rose_eternal_vigilance_policy_spawn_free_defense_platform = yes
			}
			every_owned_nonprimary_starbase = {
				limit = {
					fleet = { is_in_combat = no }
					is_controlled_by = owner
					used_defense_platform_capacity_percent < 0.33
				}
				rose_eternal_vigilance_policy_spawn_free_defense_platform = yes
			}

		}
		else_if = {
			limit = {
				has_policy_flag = rose_eternal_vigilance_policy_half_spending
			}
			every_owned_starbase = {
				limit = {
					is_starbase_type = sfortress
					fleet = { is_in_combat = no }
					is_controlled_by = owner
					used_defense_platform_capacity_percent < 0.33
				}
				rose_eternal_vigilance_policy_spawn_free_defense_platform = yes
			}
			every_owned_nonprimary_starbase = {
				limit = {
					fleet = { is_in_combat = no }
					OR = {
						is_starbase_type = sorbital_fortress
						is_starbase_type = sdeepspacecitadel
					}
					is_controlled_by = owner
					used_defense_platform_capacity_percent < 0.33
				}
				rose_eternal_vigilance_policy_spawn_free_defense_platform = yes
			}
		}
		country_event = {
			id = rose_eternal_vigilance_policy.1
			days = 180
		}
	}
}
