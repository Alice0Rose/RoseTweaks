decision_rose_natural_arcologies = {
	owned_planets_only = yes
	icon = decision_rose_natural_arcologies

	enactment_time = 3600
	resources = {
		category = decisions
		cost = {
			trigger = {
				NOT = { is_planet_class = pc_volcanic }
			}
			influence = 200
			minerals = 12500
			food = 12500
		}
		cost = {
			trigger = {
				is_planet_class = pc_volcanic
			}
			influence = 200
			minerals = 12500
			alloys = 3125
		}
	}

	potential = {
		is_potential_for_natural_arcologies = yes	
		NOT = { has_modifier = rose_natural_arcologies }
		exists = owner
		owner = { has_ascension_perk = ap_arcology_project }
	}

	allow = {
		if = {
			limit = { owner = { is_ai = no } }
			free_district_slots < 1
			custom_tooltip = {
				fail_text = legendary_leader_decision_arcology_project
				NOT = { has_planet_flag = legendary_leader_planet }
			}
		}
		custom_tooltip = {
			fail_text = not_allowed_decision_rose_natural_arcologies
			is_allowed_for_natural_arcologies = yes
		}
		
		if = {
			limit = {
				owner = { NOT = { has_civic = civic_environmentalist } }
			}
			num_uncleared_blockers < 1
		}
	}

	effect = {
		# Apply natural arcologies benefits
		custom_tooltip = decision_rose_natural_arcologies_effects
		hidden_effect = {
			add_modifier = { modifier = rose_natural_arcologies }
			
			set_planet_flag = uncapped_generator_districts
			set_planet_flag = uncapped_mining_districts
			set_planet_flag = uncapped_farming_districts
			
			relic_world_conversion_effect = {
				DISTRICT = district_city
				VARIABLE = num_housing_districts
			}
		}
	}

	ai_weight = {
		weight = 10
		modifier = {
			factor = 100
			exists = owner
			OR = {
				pop_amount >= 7000
				is_capital = yes
			}
		}
	}
}
