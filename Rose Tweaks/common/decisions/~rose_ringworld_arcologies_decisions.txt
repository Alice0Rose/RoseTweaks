decision_rose_ringworld_arcologies = {
	owned_planets_only = yes
	icon = decision_rose_ringworld_arcologies

	enactment_time = 3600
	resources = {
		category = decisions
		cost = {
			influence = 200
			alloys = 10000
		}
	}

	potential = {
		is_ringworld = yes
		NOT = { has_modifier = rose_ringworld_arcologies }
		exists = owner
		owner = {
			OR = { 
				has_ascension_perk = ap_arcology_project
				
				AND = {
					is_gestalt = yes
					has_ascension_perk = ap_master_builders
				}
			}
		}
	}

	allow = {
		if = {
			limit = { owner = { is_ai = no } }
			free_district_slots < 1
		}
	}

	effect = {
		# Apply ringworld arcologies benefits
		custom_tooltip = decision_rose_ringworld_arcologies_effects
		hidden_effect = {
			add_modifier = { modifier = rose_ringworld_arcologies }
		}
	}

	ai_weight = {
		weight = 10
		modifier = {
			factor = 100
			exists = owner
			OR = {
				pop_amount >= 7000
				is_capital = yes
			}
		}
	}
}
