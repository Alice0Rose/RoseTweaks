rose_eternal_vigilance_policy = {
	potential = {
		is_country_type = default
		has_ascension_perk = ap_eternal_vigilance
	}

	option = {
		name = rose_eternal_vigilance_policy_full_spending

		policy_flags = {
			rose_eternal_vigilance_policy_full_spending
		}

		modifier = {
			country_alloys_produces_mult = -0.05
			country_base_alloys_produces_add = -30
		}

		AI_weight = {
			weight = 20
			modifier = { #Seems like a good idea if you have the energy to spend
				factor = 5
				AND = {
					has_monthly_income = {
						resource = energy
						value > 500
					}
					has_monthly_income = {
						resource = alloys
						value > 100
					}
				}
			}
			modifier = { #Uhoh we are close to a deficit we probably shouldn't
				factor = 0
				AND = {
					has_monthly_income = {
						resource = energy
						value < 50
					}
					has_monthly_income = {
						resource = alloys
						value > 50
					}
				}
			}
		}
	}
	option = {
		name = rose_eternal_vigilance_policy_half_spending

		policy_flags = {
			rose_eternal_vigilance_policy_half_spending
		}

		modifier = {
			country_alloys_produces_mult = -0.025
			country_base_alloys_produces_add = -15
		}

		AI_weight = { #This is the option if you are doing just fine on the economy front
			weight = 75
		}
	}
	option = {
		name = rose_eternal_vigilance_policy_no_spending

		policy_flags = {
			rose_eternal_vigilance_policy_no_spending
		}

		AI_weight = {
			weight = 20
			modifier = { #Seems like a bad idea if you have the energy to spend
				factor = 0
				AND = {
					has_monthly_income = {
						resource = energy
						value > 500
					}
					has_monthly_income = {
						resource = alloys
						value > 100
					}
				}
			}
			modifier = { #Uhoh we are close to a deficit we probably shouldn't spend more
				factor = 5
				AND = {
					has_monthly_income = {
						resource = energy
						value < 50
					}
					has_monthly_income = {
						resource = alloys
						value > 50
					}
				}
			}
		}
	}
}
