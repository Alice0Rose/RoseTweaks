# Machine World Trade
col_trade_machine = {
	icon = GFX_colony_type_urban

	potential = {
		uses_district_set = machine_world
		exists = owner_or_space_owner
		owner_or_space_owner = {
			is_country_type = default
			is_machine_empire = yes
		}
		is_special_colony_type = no
		is_capital = no
		NOT = { is_planet_class = pc_nanotech }
	}

	inline_script = {
		script = colony_types/colony_type_planet_modifier
		MODIFIER = trader_jobs_bonus_workforce_mult
	}

	ai_building_set_affinity = {
		trade = 2.0
	}

	weight_modifier = {
		modifier = {
			add = @stickiness
			has_designation = col_trade_machine
		}
		complex_trigger_modifier = {
			trigger = num_districts
			parameters = {
				type = district_nexus
			}
			mode = add
			mult = 2
		}
	}
}

# Machine World Foundry
col_mac_foundry = {
	icon = GFX_colony_type_foundry

	potential = {
		uses_district_set = machine_world
		is_capital = no
		NOT = { is_planet_class = pc_nanotech }
	}

	inline_script = {
		script = colony_types/colony_type_planet_modifier
		MODIFIER = foundry_jobs_bonus_workforce_mult
	}

	triggered_planet_modifier = {
		potential = {
			owner_or_space_owner? = {
				is_galactic_community_member = yes
			}
			OR = {
				is_active_resolution = "resolution_industry_environmental_ordinance_waivers"
				is_active_resolution = "resolution_industry_project_cornucopia"
			}
		}
		modifier = {
			foundry_jobs_bonus_workforce_mult = 0.05
		}
	}

	ai_building_set_affinity = {
		foundry = 2.0
		industrial = 1.5
	}

	weight_modifier = {
		modifier = {
			add = @stickiness
			has_designation = col_mac_foundry
		}
		modifier = {
			# No mixed industry designation for machine worlds, so any foundries means this one wins
			has_any_foundry_zone = yes
			add = @zone_urban
		}
	}
}

# nanotech world
col_nanotech_world = {
	icon = GFX_colony_type_fringe

	potential = {
		is_planet_class = pc_nanotech
		owner? = { has_active_tradition = tr_nanotech_3 }
		is_special_colony_type = no
		is_capital = no
	}

	planet_modifier = {
		planet_buildings_produces_mult = 0.15
		planet_buildings_upkeep_mult = 0.1
		planet_farmers_nanites_produces_mult = 0.35
		planet_miners_nanites_produces_mult = 0.35
		planet_jobs_produces_mult = 0.05
	}

	# Steer the AI toward buildings the nanotech tradition actually unlocks: nanolabs (research/physics/
	# society/engineering sets), nanotech cauldron (industrial/foundry), and nanite transmuter
	# (industrial/factory/foundry). Without this block the AI builds generic urban infrastructure on
	# pc_nanotech worlds and never specializes them.
	ai_building_set_affinity = {
		research = 2.0
		physics = 1.5
		society = 1.5
		engineering = 1.5
		foundry = 2.0
		industrial = 1.5
		factory = 1.5
	}

	weight_modifier = {
		base = 10000
	}
}
