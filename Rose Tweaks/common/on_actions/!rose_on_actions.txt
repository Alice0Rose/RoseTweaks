on_game_start = {
	events = {
		rose_more_toxic_terraforming.1 # More toxic terraforming candidates
		rose_early_ethic_correction.1 # Allows ethics to switch to their proper distribution at the start of the game for players
	}
}

on_game_start_country = {
	events = { # country_event
		rose_fe_tech_from_planet.1 # Creates variable for number of FE planets taken
		rose_choose_starting_ruler_ethic.1 # Lets players choose their starting ruler's ethic
	}
}

# No scope, like on_game_start
on_monthly_pulse = {
	events = {
		rose_nanotech_traits.2
		rose_scaling_nanite_harvester_upgrade.1
	}
}

# No scope, like on_game_start
on_yearly_pulse = {
	events = {
		rose_nanotech_traits.3
	}
}

# this = country
on_yearly_pulse_country = {
	events = {
		paragon.6035 # Ending of "Under One Rule" event failsafe if reformists are still vibin when it triggers naturally
	}
}

# A planet has been colonized.
# Scope = Planet
on_colonized = {
	events = {
		rose_nanotech_traits.10 # Colony ship straggler
	}
}

# Planet has changed planet class in whatever way. Note that this is also called during galaxy creation in some places.
# This = Planet
on_planet_class_changed = {
	events = {
		rose_natural_arcologies.0 # Removes the Natural Arcologies modifier if a planet gets changed somehow
	}
}

# A war has ended
# Root = Loser
# From = Main Winner
on_war_ended = {
	events = {
		rose_genocidal_buffs.1
		rose_genocidal_buffs.2
		rose_genocidal_buffs.3
		rose_genocidal_buffs.4
	}
}

# A war has been won
# Root = Winner Warleader
# From = Loser Warleader
# FromFrom = War
on_war_won = {
	events = {
		rose_genocidal_buffs.100
	}
}

# Fired whenever a new owner is set for a planet, following a war or through trade
# This = Planet scope
# From = Country scope (new owner)
# FromFrom = Country scope (previous owner, if any)
on_planet_transfer = {
	events = {
		rose_fe_tech_from_planet.2 # Increments the number of FE planets a country has taken
	}
}

# A country is created via create_country or create_rebels
# This = created country
# From = root of context where create_country/create_rebels happens
on_country_created = {
	events = { # country_event
		rose_fe_tech_from_planet.1 # Creates variable for number of FE planets taken
	}
}

# Executed when a country picks any tradition (including starters and finishers)
# THIS = country
on_tradition_picked = {
	events = {
		rose_nanotech_traits.0
	}
}
